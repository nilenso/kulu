182  (defn search
183    [{:keys
184      [organization-name q name
185       from-date to-date from-submission-date to-submission-date
186       currency min-amount max-amount amount
187       status conflict expense-type
188       user-name category-name
189       order direction page per-page] :as params}]
190  
191    (let [organization-id (str (:id (org/lookup-by-name organization-name)))
192          valid-case   (fn [i] (if (str/blank? i) i (str/lower-case i)))
193          valid-number (fn [i] (if (str/blank? i) i (read-string i)))
194          order (or order "name")
195          direction (or direction "asc")
196          sort-params {(str/replace (clojure.core/name order) #"_" "-") direction}
197  
198          components [{[:term :organization-id] organization-id}
199                      {[:term :name] (valid-case name)}
200                      {[:term :status] (valid-case status)}
201                      {[:term :expense-type] (valid-case expense-type)}
202                      {[:term :currency] (valid-case currency)}
203                      {[:term :user-name] (valid-case user-name)}
204                      {[:term :category-name] (valid-case category-name)}
205                      {[:term :conflict] conflict}
206                      {[:term :amount] (valid-number amount)}
207                      {[:range :date :from] from-date [:range :date :to] to-date}
208                      {[:range :created-at :from] from-submission-date [:range :created-at :to] to-submission-date}
209                      {[:range :amount :gt] (valid-number min-amount)  [:range :amount :lt] (valid-number max-amount)}]
210  
211          build-components (remove empty? (map #(reduce-kv (fn [m k v]
212                                                             (if (nil? v)
213                                                               m (assoc-in m k v))) {} %) components))
214  
215          filters {:bool {:must build-components}}]
216      (log/infof "Searching invoices with params: %s, filters %s" params filters)
217      (let [results (inv-search/search q filters [sort-params] {:size per-page :from (Math/abs (* (dec page) per-page))})
218            total-count (:total results)
219            hits (->> (:hits results)
220                      (map (fn [res] (assoc (:_source res) :id (:_id res)))))]
221        {:meta (pagn/info {:total-count total-count
222                           :per-page per-page
223                           :page page
224                           :order order :direction direction}) :items hits})))

226  (defn export
227    [{:keys [organization-name] :as params}]
228    (let [per-page (cfg/max-export-rows)
229          results (:items (search (assoc params :per-page per-page)))
230          _ (log/infof "Search results have returned.")
231          headers [:name :date :currency :amount
232                   :expense-type :status :conflict
233                   :user-name :category-name :created-at :id]
234  
235          relevant-headers ["Merchant Name" "Expense Date" "Currency" "Amount"
236                            "Type" "Status" "Conflict"
237                            "Spender" "Category" "Submission Date" "Link"]
238  
239          parse-date-time (fn [datetime] (time-f/unparse (time-f/formatters :mysql)
240                                                         (time-f/parse (time-f/formatters :date-time-no-ms) datetime)))
241          parse-date (fn [datetime] (time-f/unparse (time-f/formatters :date)
242                                                    (time-f/parse (time-f/formatters :date-time-no-ms) datetime)))
243          relevant-rows (map #(map (-> %
244                                       (update-in [:date] parse-date)
245                                       (update-in [:created-at] parse-date-time)
246                                       (update-in [:id] inv/invoice-web-link organization-name)) headers) results)
247          data {:headers relevant-headers :rows relevant-rows}]
248      (-> data
249          (excelify/create-sheet params)
250          (excelify/send-mail params))))
