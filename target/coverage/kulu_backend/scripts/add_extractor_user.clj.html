001  (ns kulu-backend.scripts.add-extractor-user
002    (:require [kulu-backend.organizations.model :as org]))
003  
004  (defn create [email password]
005    (org/create {:user-email email
006                 :password password
007                 :user-name (str "extractor-" (str (java.util.UUID/randomUUID)))
008                 :organization-name "transcribers"}))
009  
010  ;;
011  ;; how to run:
012  ;;
013  ;; $ lein run -m kulu-backend.scripts.add-extractor-user dev sam@harris.org chomsky
014  ;;
015  (defn -main [& args]
016    (System/setProperty "nomad.env" (first args))
017    (create (second args) (first (rest (rest args))))
018    (System/exit 0))
