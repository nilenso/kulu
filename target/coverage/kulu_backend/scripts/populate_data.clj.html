001  (ns kulu-backend.scripts.populate-data
002    (:require [clojure.java.io :as io]
003              [kulu-backend.scripts.bulk-upload-invoices :as bulk-upload]))
004  
005  (defn get-files-from-dir
006    [dir-path]
007    (.listFiles (io/file dir-path)))
008  
009  (defn get-data
010    [org-name user-email dir-path]
011    (when (.isDirectory (io/file dir-path))
012      (let [files (get-files-from-dir dir-path)
013            no-files (count files)]
014        (bulk-upload/db-and-image org-name user-email
015                             (take no-files files)))))
016  
017  (defn -main [& args]
018    (System/setProperty "nomad.env" (first args))
019    (get-data (second args) (nth args 2) (nth args 3)))
