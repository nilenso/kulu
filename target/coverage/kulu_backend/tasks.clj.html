001  (ns kulu-backend.tasks
002    (:require [clojure.tools.logging :as log]
003              [kulu-backend.invoices.search :as inv-search]
004              [kulu-backend.search.common :as kulu-search]))
005  
006  (defn- migrate-indexes
007    []
008    (log/info "Migrating indexes")
009    (inv-search/migrate-indexes)
010    (System/exit 0))
011  
012  (defn- setup
013    [env]
014    (log/infof "Setting up system in environment: %s" env)
015    (System/setProperty "nomad.env" env)
016    (kulu-search/create-indexes)
017    (migrate-indexes)
018    (System/exit 0))
019  
020  (defn -main
021    "Sets up resources for the kulu backend service.
022    Commands
023    setup env       migrates DB, creates ElasticSearch indexes.
024                    env is one of production/dev/test (default: dev).
025    migrate-indexes creates ES document indexes for all invoices "
026    ([] (println (:doc (meta #'-main))))
027    ([command & args]
028       (let [[e & _] args
029             env (or e "dev")]
030         (condp = command
031           "setup" (setup env)
032           "migrate-indexes" (migrate-indexes)))))
