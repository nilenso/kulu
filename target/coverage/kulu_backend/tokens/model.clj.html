001  (ns kulu-backend.tokens.model
002    (:require [kulu-backend.redis :as r]
003              [taoensso.carmine :as car :refer (wcar)]))
004  
005  (defn- token-key
006    [key info]
007    (str (clojure.string/join ":" info) ":" (name key)))
008  
009  (defn create-token
010    [key info]
011    (let [token (str (java.util.UUID/randomUUID))]
012      (r/wcar* (car/set (token-key key info) token))
013      token))
014  
015  (defn verify-token
016    [key info token]
017    (= (str token) (r/wcar* (car/get (token-key key info)))))
018  
019  (defn expire-token
020    [key info]
021    (r/wcar* (car/expire (token-key key info) 0)))
