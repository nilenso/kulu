042        (handler req)

045  ;; TODO: this header parsing code is repeated in mailer.clj, move it to a single place.
046  (defn wrap-email-authorization [handler]
047    (fn [{:keys [params headers] :as req}]
048      (let [headers (into {} (cj/decode (:message-headers params) true))
049            from-header (headers "X-Envelope-From")
050            to-header (headers "To")
051            from (first (re-seq #"[a-zA-z0-9_.]+@[a-zA-Z0-9_.]+\.[A-z]{2,}" from-header))
052            org-name (last (first (re-seq #"
053            organization (org/lookup-by-email-and-org from org-name)]
054        (if (not (empty? organization))
055          (handler (assoc req :organization organization))
056          (unauthorized {})))))
