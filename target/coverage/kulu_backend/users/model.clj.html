001  (ns kulu-backend.users.model
002    (:require [kulu-backend.db :as db :only [query insert!]]
003              [honeysql.core :as sql]
004              [cemerick.friend.credentials :as creds]
005              [honeysql.helpers :as h-sql]))
006  
007  (defonce table-name :users)
008  
009  (defn store
010    [m]
011    (->> m
012         (db/insert! (keyword table-name))
013         first))
014  
015  (defn lookup-by-email
016    "Finds an invoice by its email."
017    [email]
018    (-> (h-sql/select :*)
019        (h-sql/from table-name)
020        (h-sql/where [:= :email email])
021        sql/format
022        db/query
023        first))
024  
025  (defn update
026    [id attrs]
027    (db/update! (keyword table-name)
028                attrs
029                ["id = ?" id])
030    id)
031  
032  (defn lookup-by-id
033    "Finds an invoice by its id."
034    [id]
035    (-> (h-sql/select :*)
036        (h-sql/from table-name)
037        (h-sql/where [:= :id id])
038        sql/format
039        db/query
040        first))
041  
042  (defn create
043    "Creates an user from the given map"
044    ([]
045     (create {}))
046    ([m]
047     (let [user (lookup-by-email (m :email))]
048       (if (empty? user)
049         (let [new-user (assoc m :password (creds/hash-bcrypt (:password m)))]
050           (store new-user))
051         user))))
052  
053  (defn update-password
054    [{:keys [id password] :as m}]
055    {:pre [(not (empty? password))]}
056    (update id {:password (creds/hash-bcrypt password)}))
